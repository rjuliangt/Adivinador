CREATE TABLE OLYMPIC_GAMES(
	ID_OLYMPIC_GAME INT NOT NULL AUTO_INCREMENT,
    NAME VARCHAR(60),
    COUNTRY VARCHAR(60),
    YEAR INT,
    CONSTRAINT PRIMARY KEY (ID_OLYMPIC_GAME)
);

CREATE TABLE DEPORTS(
	ID_DEPORT INT NOT NULL AUTO_INCREMENT,
    NAME_DEPORT VARCHAR(60),
    CATEGORIES VARCHAR(90),
    ID_OLYMPIC_GAME INT NOT NULL,
    CONSTRAINT PRIMARY KEY (ID_DEPORT),
    CONSTRAINT FOREIGN KEY DEPORT_OLYMPICGAMES (ID_OLYMPIC_GAME) REFERENCES OLYMPIC_GAMES(ID_OLYMPIC_GAME)
);

CREATE TABLE COUNTRIES(
	ID_COUNTRY INT NOT NULL AUTO_INCREMENT,
    NAME_COUNTRY VARCHAR(60),
    CITY VARCHAR(60),
    LENGUAJE VARCHAR(60),
    AREA VARCHAR(60),
    CONSTRAINT PRIMARY KEY (ID_COUNTRY)
);

CREATE TABLE PEOPLE(
	ID_PERSON INT NOT NULL AUTO_INCREMENT,
    FULL_NAME VARCHAR(60),
    AGE INT,
    GENDER CHARACTER,
    ID_COUNTRY INT,
    CONSTRAINT PRIMARY KEY (ID_PERSON),
   	CONSTRAINT FOREIGN KEY PEOPLE_COUNTRIES (ID_COUNTRY) REFERENCES COUNTRIES(ID_COUNTRY) 
);

CREATE TABLE MEDALS(
	ID_MEDAL INT NOT NULL AUTO_INCREMENT,
    NAME_MEDAL VARCHAR(60),
    COLOR VARCHAR(10),
    CONSTRAINT PRIMARY KEY (ID_MEDAL)
);

CREATE TABLE WINNERS(
	ID_WINNER INT NOT NULL AUTO_INCREMENT,
    ID_PERSON INT NOT NULL,
    ID_DEPORT INT NOT NULL,
    ID_MEDAL INT NOT NULL,
	CONSTRAINT PRIMARY KEY (ID_WINNER),
    CONSTRAINT FOREIGN KEY WINNERS_PEOPLE (ID_PERSON) REFERENCES PEOPLE(ID_PERSON),
    CONSTRAINT FOREING KEY WINNERS_DEPORTS (ID_DEPORT) REFERENCES DEPORTS(ID_DEPORT),
    CONSTRAINT FOREIGN KEY WINNERS_MEDALS (ID_MEDAL) REFERENCES MEDALS(ID_MEDAL)
);
